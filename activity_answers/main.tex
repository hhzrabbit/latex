\section{Calculations}

	\begin{enumerate}
	\item{ \[\lim_{x\to\infty} 100\] }
	\item{ \[\lim_{x\to\infty} \frac{3x^5 + 17x^4 - 18}{x^5 - 12x^2}\] }

	\end{enumerate}

	\begin{enumerate}
	\item{ \[ \int_{0}^{10} x^2 \] }
